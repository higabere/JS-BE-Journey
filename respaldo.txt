SELECT STS.ROWSECCLASS 
 , STS.SETID 
 , PD.JOBCODE 
 , DP.EFFDT 
 , PD.DESCR 
 , PSP.SAL_ADMIN_PLAN 
 , PD.GRADE 
 , PD.COMP_FREQUENCY 
 , PD.REG_TEMP 
 , DP.UPD_BY_USERID 
 , DP.LASTUPDDTTM 
 , DP.STATUS_DESCR_POS 
  FROM PS_JOBCODE_TBL PD LEFT OUTER JOIN PS_XG_DES_PUE_SIND DP ON DP.SETID = PD.SETID 
   AND DP.JOBCODE = PD.JOBCODE 
   AND DP.EFFDT = ( 
 SELECT MAX(DP_ED.EFFDT) 
  FROM PS_XG_DES_PUE_SIND DP_ED 
 WHERE DP_ED.SETID = DP.SETID 
   AND DP_ED.JOBCODE = DP.JOBCODE 
   AND DP_ED.EFFDT <= SYSDATE) , PS_XG_POS_SIN_PLAN PSP , PS_XG_SCRTY_TBL_SP STS 
 WHERE PD.EFF_STATUS = 'A' 
   AND PD.EFFDT = ( 
 SELECT MAX(PD_ED.EFFDT) 
  FROM PS_JOBCODE_TBL PD_ED  
 WHERE PD.SETID = PD_ED.SETID 
   AND PD.JOBCODE = PD_ED.JOBCODE 
   AND PD_ED.EFFDT <= SYSDATE ) 
   AND PD.SETID = PSP.SETID 
   AND PD.SAL_ADMIN_PLAN = PSP.SAL_ADMIN_PLAN 
   AND STS.SAL_ADMIN_PLAN = PSP.SAL_ADMIN_PLAN;
