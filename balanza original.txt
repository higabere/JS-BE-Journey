SELECT %P(1) 
 ,SUM(BL.POSTED_BASE_AMT) 
  FROM PS_LEDGER BL 
  , PS_ATW_BU_OPTN_CE BU 
 WHERE BL.BUSINESS_UNIT = BU.BUSINESS_UNIT 
   AND BL.LEDGER = BU.LEDGER 
   AND BU.STATUS = 'A' 
   AND BU.ATW_RFC = %P(2) 
   AND ( 
   (BL.FISCAL_YEAR = %P(3) 
   	AND BL.ACCOUNTING_PERIOD < %P(4)) 
    OR (BL.FISCAL_YEAR < %P(5))
   ) %P(6) 
  GROUP BY %P(7)
